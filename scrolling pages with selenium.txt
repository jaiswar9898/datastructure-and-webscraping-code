# how to scroll in pages with help of selenium 
from selenium import webdriver
import time
driver=webdriver.Chrome('chromedriver.exe')
driver.get('https://www.google.com/search?q=jqueryscript.net+dropdown+combo+tree&rlz=1C1SQJL_enIN933IN933&oq=&aqs=chrome.3.35i39i362l8...8.560105j0j7&sourceid=chrome&ie=UTF-8')
time.sleep(3)
previous_height=driver.execute_script('return document.body.scrollHeight')
while True:
    driver.execute_script('window.scrollTo(0,document.body.scrollHeight);')

    time.sleep(3)
    new_height=driver.execute_script('return document.body.scrollHeight')
    if new_height == previous_height:
        break
    previous_height=new_height
