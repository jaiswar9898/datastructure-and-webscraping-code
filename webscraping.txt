# how to scrape multiple pages of amazon 
from selenium import webdriver
from  bs4 import BeautifulSoup
import requests
import time
import pandas as pd
def get_html(url):
    browser = webdriver.Chrome("chromedriver.exe")
    browser.get(url)
    return browser.page_source
time.sleep(2)
def main():
        phn_nm=[]
        phn_pr=[]
        phn_ra=[]
        page_num=input("Enter the page number")  # you can take any number
        for i in range(1,int(page_num)+1):

            url=("https://www.flipkart.com/search?q=camera&otracker=search&otracker1=search&marketplace=FLIPKART&as-show=on&as=off&page"+str(i))
            html=get_html(url)
            req= requests.get(url)
            content=BeautifulSoup(req.content,'html.parser')
            name=content.find_all('div',{"class":"_4rR01T"})
            price=content.find_all('div',{"class":"_30jeq3 _1_WHN1"})
            rating=content.find_all('div',{'class':"_3LWZlK"})
            print("Phone in page"+str(i))
            #print(len(name))

            # for name
            for i in name:
                phn_nm.append(i.text)
            for i in phn_nm:
                print(i)


            for i in price:
                phn_pr.append(i.text)
            for i in phn_pr:
                print(i)


            # for rate

            for i in rating:
                phn_ra.append(i.text)
            for i in phn_ra:
                print(i)



if _name_ == '_main_':
